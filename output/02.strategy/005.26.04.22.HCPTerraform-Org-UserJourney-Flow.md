---
title: "HCP Terraform Organization User Journey Flow - ILM_Demo_Space"
---
flowchart TD
    start(["Start: Org Admin enters organization"])

    start --> access
    subgraph access["1. Access and Identity Setup"]
        direction TB
        users["Settings - Users"]
        teams["Settings - Teams"]
        auth["Settings - Authentication"]
        sso["Settings - SSO"]
        api["Settings - API Tokens"]
        ip["Settings - IP Allowlists"]
        ssh["Settings - SSH Keys"]
        users --> teams --> auth --> sso --> api --> ip --> ssh
    end

    access --> orgbase
    subgraph orgbase["2. Organization Baseline"]
        direction TB
        general["Settings - General"]
        billing["Settings - Plan and Billing"]
        tags["Settings - Tag Management"]
        health["Settings - Health"]
        general --> billing --> tags --> health
    end

    orgbase --> scm
    subgraph scm["3. Source Control and Automation Wiring"]
        direction TB
        vcsProviders["Settings - VCS Providers"]
        vcsGeneral["Settings - VCS General"]
        vcsEvents["Settings - VCS Events"]
        agents["Settings - Agents"]
        vcsProviders --> vcsGeneral --> vcsEvents --> agents
    end

    scm --> structure
    subgraph structure["4. Delivery Structure"]
        direction TB
        projects["Projects"]
        workspaces["Workspaces"]
        stacks["Stacks"]
        varSets["Settings - Variable Sets"]
        projects --> workspaces --> stacks
        varSets --> workspaces
    end

    structure --> governance
    subgraph governance["5. Governance and Guardrails"]
        direction TB
        policies["Settings - Policies"]
        policySets["Settings - Policy Sets"]
        runTasks["Settings - Run Tasks"]
        cost["Settings - Cost Estimation"]
        runs["Settings - Runs"]
        policies --> policySets --> runTasks --> cost --> runs
    end

    governance --> catalog
    subgraph catalog["6. Reuse and Discovery"]
        direction TB
        privateLib["Registry - Private Library"]
        publicNs["Registry - Public Namespaces"]
        explorer["Explorer"]
        usage["Usage"]
        privateLib --> publicNs --> explorer --> usage
    end

    catalog --> decision{"Healthy and compliant runs?"}

    decision -->|"Yes"| stable["Operate at scale\nTrack usage and expand modules"]
    decision -->|"No"| investigate

    subgraph investigate["7. Investigate and Improve"]
        direction TB
        runQueue["Settings - Runs queue and failures"]
        audit["Settings - Auditing"]
        vcsLog["Settings - VCS Events"]
        fix["Adjust policies, run tasks, variables, or auth settings"]
        runQueue --> audit --> vcsLog --> fix
    end

    investigate --> governance
    stable --> finish(["End: Governed Terraform delivery lifecycle"])
