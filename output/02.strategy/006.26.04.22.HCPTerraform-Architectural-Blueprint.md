    service engineer(server)[Platform Engineer] in users
    service auditor(server)[Auditor] in users

    group access(cloud)[Access and Identity]

    service apiTokens(server)[API Tokens] in access
    service ipAllow(server)[IP Allowlists] in access

    service workspaces(server)[Workspaces] in control
    service stacks(server)[Stacks] in control
    service varSets(database)[Variable Sets] in control
    service tags(database)[Tag Management] in control

    service vcsGeneral(server)[VCS General] in scm
    service vcsEvents(database)[VCS Events] in scm
    service agents(server)[Agent Pools] in scm
    service runs(server)[Runs] in scm

    group governance(cloud)[Governance and Policy]
    service policies(server)[Policies] in governance
    service policySets(server)[Policy Sets] in governance
    service runTasks(server)[Run Tasks] in governance
    service cost(server)[Cost Estimation] in governance
    service health(server)[Health Assessments] in governance
    service auditing(database)[Audit Trails and Streaming] in governance

    service publicNs(database)[Public Namespaces] in catalog
    service explorer(server)[Explorer] in catalog
    service usage(database)[Usage Reporting] in catalog
